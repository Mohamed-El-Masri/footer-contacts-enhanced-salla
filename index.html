⚡ كود JavaScript للتفعيل:

function enhanceFooterContacts() {
    // فحص لمنع التكرار
    const existingContainer = document.querySelector('.unified-contacts-row');
    if (existingContainer) {
        return;
    }
    
    const footerContainer = document.querySelector('.store-footer__contacts-row');
    if (!footerContainer) {
        return;
    }
    
    // إضافة علامة مؤشر لمنع إعادة التطبيق
    if (footerContainer.hasAttribute('data-enhanced')) {
        return;
    }
    
    footerContainer.setAttribute('data-enhanced', 'true');
    
    const contactItems = document.querySelectorAll('.store-footer__contacts-row .enhanced_rounded_contacts a');
    
    contactItems.forEach(function(item) {
        if (!item.classList.contains('enhanced-contact-item')) {
            item.classList.add('enhanced-contact-item');
        }
    });
    
    const footerContainer = document.querySelector('.store-footer__contacts-row');
    if (footerContainer) {
        const allContactItems = footerContainer.querySelectorAll('a');
        const socialItems = footerContainer.querySelectorAll('.social-link');
        
        const newContainer = document.createElement('div');
        newContainer.className = 'unified-contacts-row';
        newContainer.style.display = 'flex';
        newContainer.style.justifyContent = 'center';
        newContainer.style.alignItems = 'center';
        newContainer.style.flexWrap = 'wrap';
        newContainer.style.gap = '15px';
        
        allContactItems.forEach(function(item) {
            const clonedItem = item.cloneNode(true);
            clonedItem.classList.add('anime-item');
            newContainer.appendChild(clonedItem);
        });
        
        footerContainer.innerHTML = '';
        footerContainer.appendChild(newContainer);
    }
    
    const animeItems = document.querySelectorAll('.store-footer__contacts-row .anime-item');
    animeItems.forEach(function(item, index) {
        item.style.animationDelay = (index * 0.1) + 's';
    });
    
    // إيقاف Observer مؤقتاً لمنع التداخل
    if (typeof observer !== 'undefined') {
        observer.disconnect();
        setTimeout(function() {
            observer.observe(document.body, {
                childList: true,
                subtree: true
            });
        }, 1000);
    }
}

document.addEventListener('DOMContentLoaded', enhanceFooterContacts);

let observer;

// إنشاء Observer مع فحص ذكي
setTimeout(function() {
    observer = new MutationObserver(function(mutations) {
    let shouldUpdate = false;
    
    mutations.forEach(function(mutation) {
        if (mutation.type === 'childList') {
            const addedNodes = Array.from(mutation.addedNodes);
            
            // فقط إذا كان هناك عقد جديدة مضافة وليست من الكود الخاص بنا
            if (addedNodes.length > 0) {
                const hasFooterElements = addedNodes.some(node => 
                    node.nodeType === 1 && 
                    node.querySelector &&
                    (node.querySelector('.enhanced_rounded_contacts') || 
                     node.querySelector('.social-link')) &&
                    !node.querySelector('.unified-contacts-row')
                );
                
                if (hasFooterElements) {
                    shouldUpdate = true;
                }
            }
        }
    });
    
    if (shouldUpdate) {
        setTimeout(enhanceFooterContacts, 500);
    }
});

    observer.observe(document.body, {
        childList: true,
        subtree: true
    });
}, 1000);
